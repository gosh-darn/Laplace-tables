\noindent\textsf{\textbf{TABLE 2} Properties of Laplace Transforms}\\
\begin{tabular}{|C{1cm}|C{12cm}|}
\hline
1&\[\lap[Af(t)]=AF(s)\]\\
\hline
2&\[\lap[f_1(t)\pm f_2(t)]=F_1(s)\pm F_2(s)\]\\
\hline
3&\[\lap_\pm\left[\diff{}{t}f(t)\right]=sF(s)-f(0\pm)\]\\
\hline
4&\[\lap_\pm\left[\diff[2]{}{t}f(t)\right]=s^2F(s)-sf(0\pm)-\dot f(0\pm)\]\\
\hline
5&\[\lap_\pm\left[\diff[n]{}{t}f(t)\right]=s^nF(s)-\sum_{k=1}^{n}s^{n-k}f^{(k-1)}(0\pm)\]\[\text{where}\ f^{(k-1)}(t)=\diff[k-1]{}{t}f(t)\]\\
\hline
6&\[\lap_\pm\left[\int f(t)\,dt\right]=\frac{F(s)}{s}+\frac{[\int f(t)\,dt]_{t=0\pm}}{s}\]\\
\hline
7&\[\lap_\pm\left[\iint f(t)\,dt\,dt\right]=\frac{F(s)}{s^2}+\frac{[\int f(t)\,dt]_{t=0\pm}}{s^2}+\frac{[\iint f(t)\,dt\,dt]_{t=0\pm}}{s}\]\\
\hline
8&\[\lap_\pm\left[\idotsint f(t)(dt)^n\right]=\frac{F(s)}{s^n}+\sum_{k=1}^n\frac{1}{s^{n-k+1}}\left[\idotsint f(t)(dt)^k\right]_{t=0\pm}\]\\
\hline
9&\[\lap\left[\int_0^tf(t)\,dt\right]=\frac{F(s)}{s}\]\\
\hline
\end{tabular}
